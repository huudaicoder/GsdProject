Implement the automatic audit trail per D-11 and Pitfall 3 prevention.

1. **Create `AuditableEntityInterceptor.cs`** implementing `SaveChangesInterceptor`:

   ```csharp
   public class AuditableEntityInterceptor : SaveChangesInterceptor
   {
       private readonly ICurrentUserService _currentUserService;

       public AuditableEntityInterceptor(ICurrentUserService currentUserService)
       {
           _currentUserService = currentUserService;
       }

       public override ValueTask<InterceptionResult<int>> SavingChangesAsync(
           DbContextEventData eventData,
           InterceptionResult<int> result,
           CancellationToken cancellationToken = default)
       {
           var context = eventData.Context as ApplicationDbContext;
           if (context is null) return ValueTask.FromResult(result);

           var auditEntries = new List<AuditLog>();
           var userId = _currentUserService.UserId ?? "system";

           foreach (var entry in context.ChangeTracker.Entries<BaseEntity>())
           {
               if (entry.State == EntityState.Detached || entry.State == EntityState.Unchanged)
                   continue;

               var auditLog = new AuditLog
               {
                   Id = Guid.NewGuid(),
                   UserId = userId,
                   EntityType = entry.Entity.GetType().Name,
                   EntityId = entry.Entity.Id.ToString(),
                   Timestamp = DateTime.UtcNow
               };

               switch (entry.State)
               {
                   case EntityState.Added:
                       auditLog.Action = "Create";
                       entry.Entity.CreatedAt = DateTime.UtcNow;
                       auditLog.NewValues = JsonSerializer.Serialize(
                           entry.Properties.ToDictionary(p => p.Metadata.Name, p => p.CurrentValue));
                       break;

                   case EntityState.Modified:
                       auditLog.Action = "Update";
                       entry.Entity.UpdatedAt = DateTime.UtcNow;
                       auditLog.OldValues = JsonSerializer.Serialize(
                           entry.Properties.Where(p => p.IsModified)
                               .ToDictionary(p => p.Metadata.Name, p => p.OriginalValue));
                       auditLog.NewValues = JsonSerializer.Serialize(
                           entry.Properties.Where(p => p.IsModified)
                               .ToDictionary(p => p.Metadata.Name, p => p.CurrentValue));
                       break;

                   case EntityState.Deleted:
                       auditLog.Action = "Delete";
                       auditLog.OldValues = JsonSerializer.Serialize(
                           entry.Properties.ToDictionary(p => p.Metadata.Name, p => p.OriginalValue));
                       break;
               }

               auditEntries.Add(auditLog);
           }

           if (auditEntries.Any())
           {
               context.AuditLogs.AddRange(auditEntries);
           }

           return ValueTask.FromResult(result);
       }
   }
   ```

   Key points per D-11:
   - Captures: userId, action type (Create/Update/Delete), entity type, entity ID, timestamp (UTC), old values (JSON), new values (JSON)
   - Only tracks entities that inherit from `BaseEntity` (Category, Supplier, Department, and future entities)
   - Does NOT track AuditLog entries themselves (they don't inherit BaseEntity)
   - Automatically sets `CreatedAt` on Added entities and `UpdatedAt` on Modified entities

| Boundary | Description |
|----------|-------------|
| Client -> /api/auth/login | Untrusted username/password from login form |
| JWT token validation | Every request after login carries a JWT that must be validated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | LoginCommandHandler | mitigate | Use ASP.NET Core Identity `UserManager.CheckPasswordAsync()` — handles password hashing, timing-safe comparison, lockout after N failures |
| T-02-02 | Information Disclosure | Login error messages (D-10) | accept | User explicitly requested specific error messages ("username not found" vs "wrong password"). This deviates from OWASP generic-error guidance but is acceptable for admin-only internal system per user decision D-10. |
| T-02-03 | Tampering | JWT token | mitigate | Validate issuer, audience, lifetime, and signing key on every request via `TokenValidationParameters`. HS256 with 32+ char secret. |
| T-02-04 | Repudiation | Write operations | mitigate | `AuditableEntityInterceptor` logs userId + action + old/new values for every SaveChanges call. Audit logs are append-only (AuditLog entity does not inherit BaseEntity, so the interceptor does not track changes to audit logs themselves). |
| T-02-05 | Denial of Service | Login endpoint brute force | mitigate | ASP.NET Core Identity lockout: default 5 failed attempts = 5-minute lockout. Configured via `IdentityOptions.Lockout` in DI setup. |
| T-02-06 | Elevation of Privilege | JWT claims tampering | mitigate | JWT signed with HS256 server-side secret. Role claims validated server-side. FallbackPolicy requires Admin role. `[Authorize(Roles = "Admin")]` on protected endpoints for defense-in-depth. |
</threat_model>
